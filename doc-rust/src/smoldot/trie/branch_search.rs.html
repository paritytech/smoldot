18//! Allows searching for the closest branch node in a trie when only the storage trie nodes are
19//! known.
20//!
21//! This module can be used in situations where only the list of trie nodes with a storage value
22//! is known, such as in the form of a `BTreeMap`, and allows finding the list of branch nodes.
23
24use core::iter;
25
26use crate::trie;
27use alloc::vec::{IntoIter, Vec};
28
29pub use crate::trie::Nibble;
30
31mod tests;

33/// Configuration for [`start_branch_search`].
34#[derive(Debug, Clone)]
35pub struct Config<K, P> {
36    /// The search will try to find the closest node whose key is strictly superior or superior
37    /// or equal (see [`Config::or_equal`]) to the value in this field.
38    pub key_before: K,
39
40    /// If `true`, also include the node whose key is equal to [`Config::key_before`] (if any).
41    /// If `false`, only nodes whose key is strictly superior will be searched.
42    pub or_equal: bool,
43
44    /// Only search nodes whose key starts with the list of nibbles given in this field.
45    pub prefix: P,
46
47    /// If `true`, only return nodes that have a storage value associated to them.
48    ///
49    /// > **Note**: This option is provided as a convenience. The entire point of this search
50    /// >           is to find branch nodes when only storage nodes are known.
51    pub no_branch_search: bool,
52}
53
54/// Start the search algorithm.
55// TODO: maybe rename things a bit? it's weird that `NextKey` is returned
56pub fn start_branch_search(
57    config: Config<impl Iterator<Item = Nibble>, impl Iterator<Item = Nibble>>,
58) -> NextKey {
59    NextKey {
60        prefix: config.prefix.collect(),
61        key_before: config.key_before.collect(),
62        or_equal: config.or_equal,
63        inner: NextKeyInner::FirstFound {
64            no_branch_search: config.no_branch_search,
65        },
66    }
67}
68
69/// Progress in the search algorithm.
70pub enum BranchSearch {
71    /// In order to continue, the API user must indicate the trie node with a storage value that
72    /// immediately follows a given one.
73    NextKey(NextKey),
74    /// Search has finished successfully.
75    Found {
76        /// Result of the search. `None` if there is no branch/storage node after
77        /// [`Config::key_before`] in the trie.
78        branch_trie_node_key: Option<BranchTrieNodeKeyIter>,
79    },
80}
81
82/// Implementation of `Iterator<Item = Nibble>`. See [`BranchSearch::Found::branch_trie_node_key`].
83pub struct BranchTrieNodeKeyIter {
84    inner: IntoIter<Nibble>,
85}
86
87impl Iterator for BranchTrieNodeKeyIter {
88    type Item = Nibble;
89
90    fn next(&mut self) -> Option<Self::Item> {
91        self.inner.next()
92    }
93
94    fn size_hint(&self) -> (usize, Option<usize>) {
95        self.inner.size_hint()
96    }
97}
98
99impl ExactSizeIterator for BranchTrieNodeKeyIter {}

101/// In order to continue, the API user must indicate the trie node with a storage value that
102/// immediately follows a given one.
103pub struct NextKey {
104    inner: NextKeyInner,
105    /// Value passed as [`Config::key_before`].
106    key_before: Vec<Nibble>,
107    /// Value passed as [`Config::prefix`].
108    prefix: Vec<Nibble>,
109    /// Value passed as [`Config::or_equal`].
110    or_equal: bool,
111}
112
113enum NextKeyInner {
114    FirstFound {
115        /// Value passed as [`Config::no_branch_search`].
116        no_branch_search: bool,
117    },
118    FurtherRound {
119        /// Must never be empty and must never contain only `f` nibbles.
120        current_found_branch: Vec<Nibble>,
121    },
122}
123
124impl NextKey {
125    /// The API user must provide the trie node with a storage value whose key is the first one
126    /// that is strictly superior or superior or equal to the key returned by this function.
127    pub fn key_before(&self) -> impl Iterator<Item = u8> {
128        trie::nibbles_to_bytes_suffix_extend(match &self.inner {
129            NextKeyInner::FirstFound { .. } => either::Left(self.key_before.iter().copied()),
130            NextKeyInner::FurtherRound {
131                current_found_branch,
132            } => {
133                let num_f_nibbles_to_pop = current_found_branch
134                    .iter()
135                    .rev()
136                    .take_while(|n| **n == Nibble::max())
137                    .count();
138                debug_assert!(num_f_nibbles_to_pop < current_found_branch.len());
139
140                let len = current_found_branch.len();
141                let extra_nibble = current_found_branch[len - num_f_nibbles_to_pop - 1]
142                    .checked_add(1)
143                    .unwrap_or_else(|| unreachable!());
144
145                either::Right(
146                    current_found_branch
147                        .iter()
148                        .take(len - num_f_nibbles_to_pop - 1)
149                        .copied()
150                        .chain(iter::once(extra_nibble)),
151                )
152            }
153        })
154    }
155
156    /// If `true`, if the key returned by [`NextKey::key_before`] corresponds to a trie node with
157    /// a storage value, then the API user must indicate it. If `false`, it must indicate the key
158    /// strictly superior.
159    pub fn or_equal(&self) -> bool {
160        match self.inner {
161            NextKeyInner::FirstFound { .. } => {
162                // If `key_before` is for example `0xa`, `key_before()` will return `0xa0` due to
163                // the usage of `nibbles_to_bytes_suffix_extend`. For this reason, we have to
164                // always return `true` if the number of nibbles in `key_before` is uneven.
165                self.or_equal || (self.key_before.len() % 2 != 0)
166            }
167            NextKeyInner::FurtherRound { .. } => true,
168        }
169    }
170
171    /// The API user must indicate a key that starts with the bytes returned by this function.
172    /// If the first key doesn't start with these bytes, then the API user must indicate `None`.
173    pub fn prefix(&self) -> impl Iterator<Item = u8> {
174        trie::nibbles_to_bytes_truncate(self.prefix.iter().copied())
175    }
176
177    /// Indicate the key. `None` must be passed if [`NextKey::key_before`] is the last key of
178    /// the sub-trie that starts with [`NextKey::prefix`].
179    pub fn inject(
180        mut self,
181        storage_trie_node_key: Option<impl Iterator<Item = u8>>,
182    ) -> BranchSearch {
183        match (self.inner, storage_trie_node_key) {
184            // No matter what, if we're at the first round and the user injected `None`, that
185            // means `key_before` is equal or superior to the last key of the trie and thus there
186            // can't be any branch node after it.
187            (NextKeyInner::FirstFound { .. }, None) => BranchSearch::Found {
188                branch_trie_node_key: None,
189            },
190
191            // If `no_branch_search` is `true`, simply immediately return the value provided by
192            // the user.
193            (
194                NextKeyInner::FirstFound {
195                    no_branch_search: true,
196                    ..
197                },
198                Some(storage_trie_node_key),
199            ) => {
200                let storage_trie_node_key =
201                    trie::bytes_to_nibbles(storage_trie_node_key).collect::<Vec<_>>();
202                debug_assert!(storage_trie_node_key >= self.key_before);
203
204                // Due to our usage of `nibbles_to_bytes_truncate` in `prefix()`, we must check
205                // whether the value provided by the user is still within the prefix.
206                if !storage_trie_node_key.starts_with(&self.prefix) {
207                    return BranchSearch::Found {
208                        branch_trie_node_key: None,
209                    };
210                }
211
212                BranchSearch::Found {
213                    branch_trie_node_key: Some(BranchTrieNodeKeyIter {
214                        inner: storage_trie_node_key.into_iter(),
215                    }),
216                }
217            }
218
219            (
220                NextKeyInner::FirstFound {
221                    no_branch_search: false,
222                    ..
223                },
224                Some(storage_trie_node_key),
225            ) => {
226                let storage_trie_node_key =
227                    trie::bytes_to_nibbles(storage_trie_node_key).collect::<Vec<_>>();
228                debug_assert!(storage_trie_node_key >= self.key_before);
229
230                // Due to our usage of `nibbles_to_bytes_truncate` in `prefix()`, we must check
231                // whether the value provided by the user is still within the prefix.
232                if !storage_trie_node_key.starts_with(&self.prefix) {
233                    return BranchSearch::Found {
234                        branch_trie_node_key: None,
235                    };
236                }
237
238                if storage_trie_node_key.is_empty()
239                    || storage_trie_node_key.iter().all(|n| *n == Nibble::max())
240                {
241                    return BranchSearch::Found {
242                        branch_trie_node_key: Some(BranchTrieNodeKeyIter {
243                            inner: storage_trie_node_key.into_iter(),
244                        }),
245                    };
246                }
247
248                self.inner = NextKeyInner::FurtherRound {
249                    current_found_branch: storage_trie_node_key,
250                };
251                BranchSearch::NextKey(self)
252            }
253
254            (
255                NextKeyInner::FurtherRound {
256                    mut current_found_branch,
257                },
258                Some(storage_trie_node_key),
259            ) => {
260                let storage_trie_node_key = trie::bytes_to_nibbles(storage_trie_node_key);
261
262                let num_common = storage_trie_node_key
263                    .zip(current_found_branch.iter())
264                    .take_while(|(a, b)| a == *b)
265                    .count();
266
267                // Check against infinite loops.
268                debug_assert!(num_common < current_found_branch.len());
269
270                if !current_found_branch[..num_common].starts_with(&self.prefix)
271                    || &current_found_branch[..num_common] < &self.key_before
272                    || (!self.or_equal && current_found_branch[..num_common] == self.key_before)
273                {
274                    return BranchSearch::Found {
275                        branch_trie_node_key: Some(BranchTrieNodeKeyIter {
276                            inner: current_found_branch.into_iter(),
277                        }),
278                    };
279                }
280
281                current_found_branch.truncate(num_common);
282
283                if current_found_branch.is_empty()
284                    || current_found_branch.iter().all(|n| *n == Nibble::max())
285                {
286                    return BranchSearch::Found {
287                        branch_trie_node_key: Some(BranchTrieNodeKeyIter {
288                            inner: current_found_branch.into_iter(),
289                        }),
290                    };
291                }
292
293                self.inner = NextKeyInner::FurtherRound {
294                    current_found_branch,
295                };
296                BranchSearch::NextKey(self)
297            }
298
299            // If `current_found_branch` doesn't have any sibling or uncle after it, then we know
300            // that there can't be any more branch node and the search is over.
301            (
302                NextKeyInner::FurtherRound {
303                    current_found_branch,
304                },
305                None,
306            ) => BranchSearch::Found {
307                branch_trie_node_key: Some(BranchTrieNodeKeyIter {
308                    inner: current_found_branch.into_iter(),
309                }),
310            },
311        }
312    }
